# Diagnostic + safe correlation heatmap (copy-paste into a new notebook cell)
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from pathlib import Path

RESULTS_DIR = Path("../results/plots")
RESULTS_DIR.mkdir(parents=True, exist_ok=True)

# 1) show column dtypes so we can see which are numeric or not
print("Column dtypes:\n", df.dtypes)
print("\nNumeric columns ->", list(df.select_dtypes(include=[np.number]).columns))

# 2) pick numeric columns only
num_df = df.select_dtypes(include=[np.number]).copy()

if num_df.shape[1] < 2:
    print("\nNot enough numeric columns to compute a correlation matrix.")
else:
    # 3) compute correlation and plot heatmap
    corr = num_df.corr()
    print("\nCorrelation matrix:\n", corr)

    plt.figure(figsize=(8,6))
    sns.heatmap(corr, annot=True, fmt=".2f", cmap="coolwarm", square=True)
    plt.title("Correlation Heatmap (numeric cols only)")

    out_file = RESULTS_DIR / "correlation_heatmap.png"
    plt.savefig(out_file, dpi=300, bbox_inches="tight")
    plt.show()
    print(f"\nSaved heatmap to: {out_file.resolve()}")

Student Performance Analysis - Summary

Key Findings:

  1. Grade Distribution: Most students scored between 12 and 15 in final grades.
  2. Gender vs Final Grades: Both male and female students have similar performance, with no major differences.
  3. Study Time vs Final Grades: Students who dedicated more time to studying showed slightly better grades.
  4. Absences vs Final Grades: Higher absences generally led to lower grades.
  5. Correlation Heatmap: Strong positive correlation between G1, G2, and G3 (previous and final grades).

Conclusion:

  • Study time and regular attendance improve final grades.
  • Gender does not significantly impact performance.
  • Early grades (G1, G2) are good predictors of final performance (G3).
